#include <iostream>
#include <vector>

using std::cin;
using std::cout;
using std::endl;

int sum_arr(std::vector<int> arr, int len)
{
	int sum = 0;
	for (int i = 0; i < len; ++i)
	{
		if (arr[i] > 0) sum += arr[i];
	}
	return sum;
}

int main()
{
	std::vector<int> A;
	int n, num;
	cin >> n;
	for (int i = 0; i < n; ++i)
	{
		cin >> num;
		A.push_back(num);
	}
	cout << sum_arr(A, n);

	return 0;
}
